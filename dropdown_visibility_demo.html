/* User Menu Container */
.user-menu {
  position: relative;
}

/* Dropdown Menu */
.dropdown-menu {
  position: absolute;
  top: 100%;              /* Below button */
  right: 0;               /* Align right */
  margin-top: 0.75rem;    /* Spacing */
  z-index: 9999;          /* High priority */
  min-width: 200px;       /* Readability */
  
  /* Hidden State */
  opacity: 0;
  visibility: hidden;
  transform: translateY(-10px);
  pointer-events: none;
}

/* Visible State on Hover */
.user-menu:hover .dropdown-menu {
  opacity: 1;
  visibility: visible;
  transform: translateY(0);
  pointer-events: auto;
}
